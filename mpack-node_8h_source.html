109 /*
110  * mpack_tree_link_t forms a linked list of node pages. It is allocated
111  * separately from the page so that we can store the first link internally
112  * without a malloc (the only link in a pooled tree), and we don't
113  * affect the size of page pools or violate strict aliasing.
114  */
115 typedef struct mpack_tree_link_t {
116  struct mpack_tree_link_t* next;
117  mpack_node_data_t* nodes;
118  size_t pos;
119  size_t left;
120 } mpack_tree_link_t;
