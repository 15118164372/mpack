353 static inline mpack_tag_t mpack_node_tag(mpack_node_t node) {
354  mpack_tag_t tag;
355  mpack_memset(&tag, 0, sizeof(tag));
356  tag.type = node.data->type;
357  switch (node.data->type) {
358  case mpack_type_nil: break;
359  case mpack_type_bool: tag.v.b = node.data->value.b; break;
360  case mpack_type_float: tag.v.f = node.data->value.f; break;
361  case mpack_type_double: tag.v.d = node.data->value.d; break;
362  case mpack_type_int: tag.v.i = node.data->value.i; break;
363  case mpack_type_uint: tag.v.u = node.data->value.u; break;
364  case mpack_type_str: tag.v.l = node.data->value.data.l; break;
365  case mpack_type_bin: tag.v.l = node.data->value.data.l; break;
366  case mpack_type_ext: tag.v.l = node.data->value.data.l; break;
367  case mpack_type_array: tag.v.n = node.data->value.content.n; break;
368  case mpack_type_map: tag.v.n = node.data->value.content.n; break;
369  }
370  return tag;
371 }
