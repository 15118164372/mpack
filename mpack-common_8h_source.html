333 static inline mpack_error_t mpack_track_push(mpack_track_t* track, mpack_type_t type, uint64_t count) {
334  mpack_assert(track->elements, "null track elements!");
335 
336  // maps have twice the number of elements (key/value pairs)
337  if (type == mpack_type_map)
338  count *= 2;
339 
340  // grow if needed
341  if (track->count == track->capacity) {
342  mpack_error_t error = mpack_track_grow(track);
343  if (error != mpack_ok)
344  return error;
345  }
346 
347  // insert new track
348  track->elements[track->count].type = type;
349  track->elements[track->count].left = count;
350  ++track->count;
351  return mpack_ok;
352 }

354 static inline mpack_error_t mpack_track_pop(mpack_track_t* track, mpack_type_t type) {
355  mpack_assert(track->elements, "null track elements!");
356 
357  if (track->count == 0) {
358  mpack_break("attempting to close a %s but nothing was opened!", mpack_type_to_string(type));
359  return mpack_error_bug;
360  }
361 
362  mpack_track_element_t* element = &track->elements[track->count - 1];
363 
364  if (element->type != type) {
365  mpack_break("attempting to close a %s but the open element is a %s!",
366  mpack_type_to_string(type), mpack_type_to_string(element->type));
367  return mpack_error_bug;
368  }
369 
370  if (element->left != 0) {
371  mpack_break("attempting to close a %s but there are %" PRIu64 " %s left",
372  mpack_type_to_string(type), element->left,
373  (type == mpack_type_map || type == mpack_type_array) ? "elements" : "bytes");
374  return mpack_error_bug;
375  }
376 
377  --track->count;
378  return mpack_ok;
379 }

381 static inline mpack_error_t mpack_track_element(mpack_track_t* track, bool read) {
382  MPACK_UNUSED(read);
383  mpack_assert(track->elements, "null track elements!");
384 
385  // if there are no open elements, that's fine, we can read elements at will
386  if (track->count == 0)
387  return mpack_ok;
388 
389  mpack_track_element_t* element = &track->elements[track->count - 1];
390 
391  if (element->type != mpack_type_map && element->type != mpack_type_array) {
392  mpack_break("elements cannot be %s within an %s", read ? "read" : "written",
393  mpack_type_to_string(element->type));
394  return mpack_error_bug;
395  }
396 
397  if (element->left == 0) {
398  mpack_break("too many elements %s for %s", read ? "read" : "written",
399  mpack_type_to_string(element->type));
400  return mpack_error_bug;
401  }
402 
403  --element->left;
404  return mpack_ok;
405 }

407 static inline mpack_error_t mpack_track_bytes(mpack_track_t* track, bool read, uint64_t count) {
408  MPACK_UNUSED(read);
409  mpack_assert(track->elements, "null track elements!");
410 
411  if (track->count == 0) {
412  mpack_break("bytes cannot be %s with no open bin, str or ext", read ? "read" : "written");
413  return mpack_error_bug;
414  }
415 
416  mpack_track_element_t* element = &track->elements[track->count - 1];
417 
418  if (element->type == mpack_type_map || element->type == mpack_type_array) {
419  mpack_break("bytes cannot be %s within an %s", read ? "read" : "written",
420  mpack_type_to_string(element->type));
421  return mpack_error_bug;
422  }
423 
424  if (element->left < count) {
425  mpack_break("too many bytes %s for %s", read ? "read" : "written",
426  mpack_type_to_string(element->type));
427  return mpack_error_bug;
428  }
429 
430  element->left -= count;
431  return mpack_ok;
432 }

487 static inline
488 uint32_t mpack_utf8_decode(uint32_t* state, uint32_t* codep, uint32_t byte) {
489  uint32_t type = mpack_utf8d[byte];
490 
491  *codep = (*state != MPACK_UTF8_ACCEPT) ?
492  (byte & 0x3fu) | (*codep << 6) :
493  (0xff >> type) & (byte);
494 
495  *state = mpack_utf8d[256 + *state + type];
496  return *state;
497 }
