336 /*
337  * Helpers to perform unaligned network-endian loads and stores
338  * at arbitrary addresses.
339  *
340  * These will remain available in the public API so feel free to
341  * use them for other purposes, but they are undocumented. (Note
342  * also that they are static always-inline; they do not follow
343  * the normal MPack inline linkage.)
344  *
345  * The bswap builtins are used when needed and available. With
346  * GCC 5.2 they appear to give better performance and smaller
347  * code size on little-endian ARM while compiling to the same
348  * assembly as the bit-shifting code on x86_64.
349  */

351 MPACK_ALWAYS_INLINE uint8_t mpack_load_native_u8(const char* p) {
352  return (uint8_t)p[0];
353 }

355 MPACK_ALWAYS_INLINE uint16_t mpack_load_native_u16(const char* p) {
356  return (uint16_t)((((uint16_t)(uint8_t)p[0]) << 8) |
357  ((uint16_t)(uint8_t)p[1]));
358 }

360 MPACK_ALWAYS_INLINE uint32_t mpack_load_native_u32(const char* p) {
361  #ifdef MPACK_NHSWAP32
362  uint32_t val;
363  mpack_memcpy(&val, p, sizeof(val));
364  return MPACK_NHSWAP32(val);
365  #else
366  return (((uint32_t)(uint8_t)p[0]) << 24) |
367  (((uint32_t)(uint8_t)p[1]) << 16) |
368  (((uint32_t)(uint8_t)p[2]) << 8) |
369  ((uint32_t)(uint8_t)p[3]);
370  #endif
371 }

373 MPACK_ALWAYS_INLINE uint64_t mpack_load_native_u64(const char* p) {
374  #ifdef MPACK_NHSWAP64
375  uint64_t val;
376  mpack_memcpy(&val, p, sizeof(val));
377  return MPACK_NHSWAP64(val);
378  #else
379  return (((uint64_t)(uint8_t)p[0]) << 56) |
380  (((uint64_t)(uint8_t)p[1]) << 48) |
381  (((uint64_t)(uint8_t)p[2]) << 40) |
382  (((uint64_t)(uint8_t)p[3]) << 32) |
383  (((uint64_t)(uint8_t)p[4]) << 24) |
384  (((uint64_t)(uint8_t)p[5]) << 16) |
385  (((uint64_t)(uint8_t)p[6]) << 8) |
386  ((uint64_t)(uint8_t)p[7]);
387  #endif
388 }

401 MPACK_ALWAYS_INLINE void mpack_store_native_u32(char* p, uint32_t val) {
402  #ifdef MPACK_NHSWAP32
403  val = MPACK_NHSWAP32(val);
404  mpack_memcpy(p, &val, sizeof(val));
405  #else
406  uint8_t* u = (uint8_t*)p;
407  u[0] = (uint8_t)((val >> 24) & 0xFF);
408  u[1] = (uint8_t)((val >> 16) & 0xFF);
409  u[2] = (uint8_t)((val >> 8) & 0xFF);
410  u[3] = (uint8_t)( val & 0xFF);
411  #endif
412 }

414 MPACK_ALWAYS_INLINE void mpack_store_native_u64(char* p, uint64_t val) {
415  #ifdef MPACK_NHSWAP64
416  val = MPACK_NHSWAP64(val);
417  mpack_memcpy(p, &val, sizeof(val));
418  #else
419  uint8_t* u = (uint8_t*)p;
420  u[0] = (uint8_t)((val >> 56) & 0xFF);
421  u[1] = (uint8_t)((val >> 48) & 0xFF);
