475 // Reads count bytes into p, deferring to mpack_read_native_big() if more
476 // bytes are needed than are available in the buffer.
477 static inline void mpack_read_native(mpack_reader_t* reader, char* p, size_t count) {
478  if (count > reader->left) {
479  mpack_read_native_big(reader, p, count);
480  } else {
481  mpack_memcpy(p, reader->buffer + reader->pos, count);
482  reader->pos += count;
483  reader->left -= count;
484  }
485 }
486 
487 // Reads native bytes with jump disabled. This allows mpack reader functions
488 // to hold an allocated buffer and read native data into it without leaking it.
489 static inline void mpack_read_native_nojump(mpack_reader_t* reader, char* p, size_t count) {
490  #if MPACK_SETJMP
491  jmp_buf* jump_env = reader->jump_env;
492  reader->jump_env = NULL;
493  #endif
494  mpack_read_native(reader, p, count);
495  #if MPACK_SETJMP
496  reader->jump_env = jump_env;
497  #endif
498 }

500 MPACK_ALWAYS_INLINE uint8_t mpack_read_native_u8(mpack_reader_t* reader) {
501  if (reader->left >= sizeof(uint8_t)) {
502  uint8_t ret = mpack_load_native_u8(reader->buffer + reader->pos);
503  reader->pos += sizeof(uint8_t);
504  reader->left -= sizeof(uint8_t);
505  return ret;
506  }
